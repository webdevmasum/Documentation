Example 3: whereIn() multiple condition
>>>>>>>>>><<<<<<<<<>>>>>>>>>><<<<<<<<<

    use App\Models\User;

    public function getSelectedUsers()
    {
        $users = User::whereIn('status', ['active', 'pending'])->get();

        return view('users.index', compact('users'));
    }
